import pandas as pd
from sklearn.cluster import DBSCAN
import matplotlib.pyplot as plt
import seaborn as sns

iris = sns.load_dataset('iris')
iris_features = iris.drop(columns=['species'])

dbscan_model = DBSCAN(eps=0.45, min_samples=5)
clusters = dbscan_model.fit_predict(iris_features)

ct = pd.crosstab(clusters, iris['species'])
print(ct)

plt.scatter(iris_features.iloc[:,0], iris_features.iloc[:,1], c=clusters)
plt.title('DBSCAN Clustering')
plt.show()
