import cv2
import numpy as np
import matplotlib.pyplot as plt

n = int(input('Enter the input samples: '))

img_path = '/content/vegeta.jpg'
img_bgr = cv2.imread(img_path)

if img_bgr is None:
    print(f"Error: Could not load image from {img_path}")
else:
    img_rgb = cv2.cvtColor(img_bgr, cv2.COLOR_BGR2RGB)

    h, w, c = img_rgb.shape
    im_color = np.zeros_like(img_rgb, dtype=np.uint8)

    for i in range(0, h, n):
        for j in range(0, w, n):
            block_color = img_rgb[i, j] 
            for k_offset in range(n):
                for l_offset in range(n):
                    if i + k_offset < h and j + l_offset < w:
                        im_color[i + k_offset, j + l_offset] = block_color

    plt.figure(figsize=(10, 5))
    plt.subplot(1, 2, 1)
    plt.imshow(img_rgb)
    plt.title('Original RGB Image')
    plt.axis('off')

    plt.subplot(1, 2, 2)
    plt.imshow(im_color)
    plt.title(f'Sampled Color Image (n={n})')
    plt.axis('off')

    plt.show()
